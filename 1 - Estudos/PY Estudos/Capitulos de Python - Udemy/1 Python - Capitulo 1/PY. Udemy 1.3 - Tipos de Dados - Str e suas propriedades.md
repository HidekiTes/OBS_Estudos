```python
# Tipos de Dados: Str e suas propriedades

"""
Python = Linguagem de programação
Tipo de tipagem = Dinâmica / Forte
str -> string -> texto
Strings são textos que estão dentro de aspas
"""

print(1234)

# Aspas simples
print('Pedro Luiz')
print(1,'Pedro "Luiz"')

# Aspas duplas
print("Pedro Luiz")
print(2,"Pedro 'Luiz'")

# Escape
print("Pedro \"Luiz\" ")

# r
print(r"Pedro \"Luiz\" ")
```
