```python
'''  
Demonstração de Operadores Aritméticos  
(Adição - Subtração - Multiplicação - Divisão - Divisão Inteira - Exponenciação Módulo)  
'''  
  
adicao = 10 + 10  
print('Adição', adicao)  
  
subtracao = 10 - 5  
print('Subtração', subtracao)  
  
multiplicacao = 10 * 10  
print('Multiplicação', multiplicacao)  
  
divisao = 10 / 2.2 # Retorno dos valores sempre em Float  
print('Divisao', divisao)  
  
divisao_inteira = 10 // 2.2  
print('Divisão inteira', divisao_inteira)  
  
exponenciacao = 2 ** 10  
print('Exponenciação', exponenciacao)  
  
modulo = 55 % 2 #resto da divisao  
print('Módulo', modulo)  
  
print(10 % 8 == 0) # Verificacao se o número é divisível por 8  
print(16 % 8 == 0) # Verificacao se o número é divisível por 8  
print(15 % 2 == 0) # Verificacao se o número é par  
print(16 % 2 == 0) # Verificacao se o número é par
```
