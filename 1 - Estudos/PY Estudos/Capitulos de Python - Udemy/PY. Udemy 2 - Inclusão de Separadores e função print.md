```python
# Inclusão de Separadores e Função Print
# Sep é utilizado como separador entre os argumentos dentro do print
# \r\n - Quebra a linha no CRLF
# \n - Quebra a linha no LF

print(12,34,1011, sep='|', end='\n##')
print(56,78, sep='-',end='\n')
print(9,10, sep='-',end='\n')
```
